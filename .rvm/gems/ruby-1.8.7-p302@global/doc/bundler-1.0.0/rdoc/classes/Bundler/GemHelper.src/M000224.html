    def install_gem
      built_gem_path = build_gem
      out, err, code = sh_with_code("gem install #{built_gem_path}")
      if err[/ERROR/]
        Bundler.ui.error err
      else
        Bundler.ui.confirm "#{name} (#{version}) installed"
      end
    end
